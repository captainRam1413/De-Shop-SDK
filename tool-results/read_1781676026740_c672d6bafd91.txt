   146→   146→/** Coerce an SDK Asset.rarity string into our 4-tier UI Rarity (defensive). */
   147→   147→function rarityFromAsset(raw: string): Rarity {
   148→   148→  const r = (raw || '').toLowerCase()
   149→   149→  if (r === 'common' || r === 'rare' || r === 'epic' || r === 'legendary') return r
   150→   150→  // Map the SDK's expanded rarities (uncommon, mythic) down to the nearest UI tier.
   151→   151→  if (r === 'uncommon') return 'common'
   152→   152→  if (r === 'mythic') return 'legendary'
   153→   153→  return 'common'
   154→   154→}

   156→   156→/** Pick a Nexus Forge-style icon emoji for the listing based on rarity + skin type. */
   157→   157→function iconForAsset(rarity: Rarity, skinType?: string): string {
   158→   158→  if (skinType === 'character') return '🧑'
   159→   159→  if (skinType === 'accessory') return '✨'
   160→   160→  // Default: weapon icon, tinted by rarity tier.
   161→   161→  switch (rarity) {
   162→   162→    case 'legendary': return '⚔️'
   163→   163→    case 'epic':      return '🗡️'
   164→   164→    case 'rare':      return '💎'
   165→   165→    default:          return '🪨'
   166→   166→  }
   167→   167→}
   168→   168→
   169→   169→/**
   170→   170→ * Build a 7-day price-history array for the chart. Without per-asset historical
   171→   171→ * pricing data from the backend we render a flat line at the current list price
   172→   172→ * (volatility=0 → trend shows "—"). When `volatility` is provided, a stable
   173→   173→ * pseudo-random series is generated so the chart has visible variation; this is
   174→   174→ * only used for visual continuity, NOT for fake market data.
   175→   175→ */
   176→   176→function generatePriceHistory(basePrice: number, volatility: number = 0): PricePoint[] {
   177→   177→  const days = ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun']
   178→   178→  let current = basePrice
   179→   179→  return days.map((day) => {
   180→   180→    if (volatility > 0) {
   181→   181→      const change = (Math.random() - 0.45) * volatility * basePrice
   182→   182→      current = Math.max(basePrice * 0.5, current + change)
   183→   183→    }
   184→   184→    return { day, price: Math.round(current * 100) / 100 }
   185→   185→  })
   186→   186→}
   187→   187→
   188→   188→/** Format an ISO timestamp as a short relative-time string ("3h ago", "2d ago"). */
   189→   189→function formatRelativeTime(iso: string): string {
   190→   190→  const then = new Date(iso).getTime()
   191→   191→  if (Number.isNaN(then)) return 'recently'
   192→   192→  const diffMs = Date.now() - then
   193→   193→  const m = Math.floor(diffMs / 60_000)
   194→   194→  if (m < 1) return 'just now'
   195→   195→  if (m < 60) return `${m}m ago`
   196→   196→  const h = Math.floor(m / 60)
   197→   197→  if (h < 24) return `${h}h ago`
   198→   198→  const d = Math.floor(h / 24)
   199→   199→  return `${d}d ago`
   200→   200→}

   202→   202→/** Build a short seller label from a wallet address (first 8 chars + ellipsis). */
   203→   203→function shortSeller(wallet: string): string {
   204→   204→  if (!wallet) return 'unknown'
   205→   205→  return wallet.length > 12 ? `${wallet.slice(0, 8)}…` : wallet
   206→   206→}

   208→   208→/** Default description shown in the detail modal — derived from rarity + type. */
   209→   209→function describeAsset(name: string, rarity: Rarity, type?: string): string {
   210→   210→  const tier = rarity === 'legendary'
   211→   211→    ? 'a legendary-tier'
   212→   212→    : rarity === 'epic'
   213→   213→      ? 'an epic-tier'
   214→   214→      : rarity === 'rare'
   215→   215→        ? 'a rare-tier'
   216→   216→        : 'a common-tier'
   217→   217→  const kind = type === 'character' ? 'character skin' : type === 'accessory' ? 'accessory' : 'weapon skin'
   218→   218→  return `${name} is ${tier} ${kind} minted on Algorand and listed on the De-Shop marketplace.`
   219→   219→}

   221→   221→/** Convert a real SDK Asset into the UI's MarketplaceListing shape. */
   222→   222→function assetToListing(asset: Asset): MarketplaceListing {
   223→   223→  const rarity = rarityFromAsset(asset.rarity)
   224→   224→  const skinType = asset.metadata?.skin_type
   225→   225→  const price = asset.list_price ?? 0
   226→   226→  const suggested = asset.suggested_price
   227→   227→  return {
   228→   228→    id: String(asset.id),
   229→   229→    assetId: asset.id,
   230→   230→    asaId: asset.asa_id,
   231→   231→    name: asset.name,
   232→   232→    rarity,
   233→   233→    price,
   234→   234→    seller: shortSeller(asset.owner || asset.creator),
   235→   235→    icon: iconForAsset(rarity, skinType),
   236→   236→    priceHistory: generatePriceHistory(price || 1),
   237→   237→    confidence: suggested?.confidence ?? 0,
   238→   238→    suggestedPrice: suggested?.price ?? price,
   239→   239→    rarityScore: suggested?.rarity_score ?? 0,
   240→   240→    listedAt: formatRelativeTime(asset.created_at),
   241→   241→    type: skinType ?? 'item',
   242→   242→    description: describeAsset(asset.name, rarity, skinType),
   243→   243→  }
   244→   244→}

   246→   246→/** Compute floor price + listing-count per rarity from a list of listings. */
   247→   247→function computeFloorPrice(listings: MarketplaceListing[]): Record<Rarity, { price: number; change24h: number; listCount: number }> {
   248→   248→  const empty = { price: 0, change24h: 0, listCount: 0 }
   249→   249→  const out: Record<Rarity, { price: number; change24h: number; listCount: number }> = {
   250→   250→    common: { ...empty },
   251→   251→    rare: { ...empty },
   252→   252→    epic: { ...empty },
   253→   253→    legendary: { ...empty },
   254→   254→  }
   255→   255→  for (const l of listings) {
   256→   256→    const bucket = out[l.rarity]
   257→   257→    bucket.listCount += 1
   258→   258→    if (bucket.price === 0 || (l.price > 0 && l.price < bucket.price)) {
   259→   259→      bucket.price = l.price
   260→   260→    }
   261→   261→  }
   262→   262→  // 24h change requires historical data the backend doesn't currently expose;
   263→   263→  // leave at 0 so the UI shows "—" rather than fabricated percentages.
   264→   264→  return out
   265→   265→}
